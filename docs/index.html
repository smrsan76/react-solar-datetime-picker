Usage
Inline Mode
import React, {Component} from 'react';
import moment from 'moment-jalaali';

import "react-solar-datetime-picker/dist/react-solar-datetime-picker.css";
import {DatetimePicker} from 'react-solar-datetime-picker';

class App extends Component {
  constructor() {
    super();
    this.state = {
      moment: moment()
    };
  }

  handleChange = (moment) => {
    this.setState({
      moment
    });
  }

  render() {
    return (
      <DatetimePicker
        splitPanel={true}
        moment={this.state.moment}
        onChange={this.handleChange}
        showTimePicker={false}
      />
    );
  }
}

render(
  <App />,
  document.getElementById('app')
);
Popup Mode
import React, {Component} from 'react';
import moment from 'moment-jalaali';

import "react-solar-datetime-picker/dist/react-solar-datetime-picker.css";
import {DatetimePickerTrigger} from 'react-solar-datetime-picker';

class App extends Component {
  constructor() {
    super();
    this.state = {
      moment: moment()
    };
  }

  handleChange = (moment) => {
    this.setState({
      moment
    });
  }

  render() {
    const shortcuts = {
      'Today': moment(),
      'Yesterday': moment().subtract(1, 'days'),
      'Clear': ''
    };
    
    return (
      <DatetimePickerTrigger
        shortcuts={shortcuts} 
        moment={this.state.moment}
        onChange={this.handleChange}
        showTimePicker={false}>
        <input type="text" value={this.state.moment.format('YYYY-MM-DD')} readOnly />
      </DatetimePickerTrigger>
    );
  }
}

render(
  <App />,
  document.getElementById('app')
);

You can also popup the datetime picker to the top position. Everything is similar to the codes above, but just add a 'position' attribute with 'top' value to the 'DatetimePickerTrigger' component. 

...
<DatetimePickerTrigger
  ...
  position="top">
  ...
</DatetimePickerTrigger>
